\begin{figure*}[t]
\centering
\begin{tikzpicture}[
    node distance=0.8cm and 0.6cm,
    phase/.style={draw, rounded corners, minimum width=2.2cm, minimum height=0.7cm,
                  font=\footnotesize, align=center},
    layer1/.style={phase, fill=blue!8},
    layer2/.style={phase, fill=orange!10},
    layer3/.style={phase, fill=green!8},
    arr/.style={-{Stealth[length=2mm]}, thick, gray!70},
    layerbox/.style={draw=gray!40, dashed, rounded corners=3pt, inner sep=6pt},
    layerlabel/.style={font=\scriptsize\sffamily, gray!60},
]

% Layer 1: Knowledge Acquisition
\node[layer1] (model)   {Model\\Analysis};
\node[layer1, right=1.2cm of model] (doc) {Document\\Analysis};
\node[layer1, right=1.2cm of doc] (pattern) {Pattern\\Learning};

% Layer 1 internal dependency: pattern learning needs model analysis
\draw[arr] (model.east) -- (pattern.west);

% Layer 2: Link Recovery
\node[layer2, below=1.4cm of model, xshift=-0.8cm] (explicit) {Explicit\\Ref.\ Extr.};
\node[layer2, right=0.6cm of explicit] (contextual) {Contextual\\Ref.\ Disc.};
\node[layer2, right=0.6cm of contextual] (anaphoric) {Anaphoric\\Ref.\ Res.};
\node[layer2, right=0.6cm of anaphoric] (abbreviated) {Abbreviated\\Ref.\ Match};

% Knowledge flow arrows (Layer 1 -> Layer 2)
\draw[arr] (model.south) -- (contextual.north);
\draw[arr] (model.south) -- (anaphoric.north);
\draw[arr] (doc.south) -- (contextual.north);
\draw[arr] (doc.south) -- (anaphoric.north);
\draw[arr] (doc.south) -- (abbreviated.north);
\draw[arr] (pattern.south) -- (contextual.north);
\draw[arr] (pattern.south) -- (anaphoric.north);

% Layer 3: Link Consolidation
\node[layer3, below=1.4cm of contextual, xshift=0.8cm] (merge) {Priority Merge};
\node[layer3, below=0.5cm of merge] (boundary) {Boundary Filter};
\node[layer3, below=0.5cm of boundary] (evidence) {Evidence Filter};

% Layer 2 -> Layer 3
\draw[arr] (explicit.south) -- (merge.north west);
\draw[arr] (contextual.south) -- (merge.north);
\draw[arr] (anaphoric.south) -- (merge.north);
\draw[arr] (abbreviated.south) -- (merge.north east);

\draw[arr] (merge) -- (boundary);
\draw[arr] (boundary) -- (evidence);

% Output
\node[below=0.4cm of evidence, font=\footnotesize\itshape] (out) {Trace Links $L$};
\draw[arr] (evidence) -- (out);

% Layer labels
\node[layerlabel, left=0.3cm of model] {Layer 1};
\node[layerlabel, left=0.3cm of explicit] {Layer 2};
\node[layerlabel, left=0.8cm of merge] {Layer 3};

% Layer boxes
\begin{scope}[on background layer]
    \node[layerbox, fit=(model)(pattern), label={[layerlabel]above:Knowledge Acquisition}] {};
    \node[layerbox, fit=(explicit)(abbreviated), label={[layerlabel]above left:Link Recovery}] {};
    \node[layerbox, fit=(merge)(evidence)] {};
\end{scope}

\end{tikzpicture}
\caption{Pipeline architecture as a staged DAG. Layer~1 acquires knowledge
concurrently; pattern learning depends on model analysis results.
Layer~2 runs four reference-type-specific recovery strategies; explicit
reference extraction has no knowledge dependencies and starts immediately.
Layer~3 merges all candidates and applies boundary and evidence filtering.}
\label{fig:pipeline}
\end{figure*}
